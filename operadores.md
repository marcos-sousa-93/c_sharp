### 3. Operadores
```csharp
// Aritméticos
int soma = 5 + 3;        // 8
int subtracao = 5 - 3;   // 2
int multiplicacao = 5 * 3; // 15
int divisao = 15 / 3;    // 5
int resto = 5 % 3;       // 2

// Comparação
bool igual = (5 == 3);   // false
bool diferente = (5 != 3); // true
bool maior = (5 > 3);    // true
bool menor = (5 < 3);    // false

// Lógicos
bool and = (true && false);  // false
bool or = (true || false);   // true
bool not = !true;            // false
```
